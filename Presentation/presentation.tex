\begin{frame}{Problemstellung}
\begin{itemize}
\item Computer allgegenwärtig
	\begin{itemize}
	\item[$\Rightarrow$] mobile Endgeräte
	\item[$\Rightarrow$] "`ubiquitous computing"'
	\end{itemize}
\item Positionsbestimmung relativ genau möglich
\end{itemize}

\begin{itemize}
\item wirtschaftlicher Nutzen?
	\begin{itemize}
	\item[$\Rightarrow$] positionsabhängige, personalisierte Dienste
	\item[$\Rightarrow$] z.B. Mittagskarte eines Restaurants in der Nähe
	\end{itemize}
\end{itemize}

\begin{itemize}
\item Positionsvorhersage statt Positionsbestimmung?
	\begin{itemize}
	\item[$\Rightarrow$] positionsabhängige Dienste im Voraus bereitstellen
	\item[$\Rightarrow$] Mittagskarte des Restaurants, wenn ich mich gerade auf dem Weg dorthin befinde
	\end{itemize}
\end{itemize}

\begin{frame}{Aufbau einer Applikation}
\begin{itemize}
\item die Activity übernimmt Darstellung des User-Interface sowie Ausführung von Logik
\item ein Service übernimmt Aufgaben im Hintergrund
\item Intents sind abstrakte Beschreibungen von Operationen, die eine App benötigt und eine andere anbieten kann
\item BroadcastReceiver warten auf Intents und verarbeitet diese
\item AndroidManifest enthält Informationen darüber welche Android-Versionen untersützt werden, Versionen der App, welche Berechtigungen nötig sind
\item darüber hinaus wird ein Layout-XML benötigt
\end{itemize}
\end{frame}
